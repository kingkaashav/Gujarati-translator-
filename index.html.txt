  <script>
    // Simple dictionary: English → { romanGujarati, gujaratiScript }
    // You can keep adding more words here.
    const dictionary = {
      "hello": { roman: "namaste", gu: "નમસ્તે" },
      "how are you": { roman: "kem chho", gu: "કેમ છો" },
      "good morning": { roman: "shubh prabhat", gu: "શુભ પ્રભાત" },
      "thank you": { roman: "aabhar", gu: "આભાર" },
      "yes": { roman: "haa", gu: "હા" },
      "no": { roman: "naa", gu: "ના" },
      "please": { roman: "krupaa kari", gu: "કૃપા કરી" },
      "good night": { roman: "shubh raatri", gu: "શુભ રાત્રિ" }
    };

    // Very simple phrase matcher:
    // 1. Lowercase input
    // 2. Try to match known phrases
    // 3. If not found, just echo back with a note
    function translatePhrase(english) {
      const cleaned = english.trim().toLowerCase();

      // Exact phrase match
      if (dictionary[cleaned]) {
        return dictionary[cleaned];
      }

      // Try to match inside sentence (e.g. "hello, how are you?")
      for (const key of Object.keys(dictionary)) {
        if (cleaned.includes(key)) {
          return dictionary[key];
        }
      }

      // Fallback: no translation yet
      return {
        roman: "(no saved translation yet for: \"" + english + "\")",
        gu: "(અહીં સુધી અનુવાદ ઉમેર્યો નથી)"
      };
    }

    document.getElementById("translateBtn").addEventListener("click", () => {
      const englishInput = document.getElementById("englishInput").value;
      const result = translatePhrase(englishInput);

      document.getElementById("romanOutput").textContent = result.roman;
      document.getElementById("gujaratiOutput").textContent = result.gu;
    });
  </script>
