<script>
// Basic transliteration map (English → Gujarati sounds)
const map = {
  a:"અ", aa:"આ", i:"ઇ", ee:"ઈ", u:"ઉ", oo:"ઊ",
  e:"એ", ai:"ઐ", o:"ઓ", au:"ઔ",
  k:"ક", kh:"ખ", g:"ગ", gh:"ઘ",
  ch:"ચ", j:"જ", t:"ટ", th:"થ", d:"ડ", dh:"ઢ",
  n:"ન", p:"પ", ph:"ફ", b:"બ", bh:"ભ",
  m:"મ", y:"ય", r:"ર", l:"લ", v:"વ",
  sh:"શ", s:"સ", h:"હ"
};

// Roman Gujarati generator
function toRomanGujarati(text) {
  return text
    .toLowerCase()
    .replace(/th/g, "t")
    .replace(/ph/g, "f")
    .replace(/sh/g, "sh")
    .replace(/ch/g, "ch")
    .replace(/kh/g, "kh")
    .replace(/bh/g, "bh");
}

// Gujarati script generator (very simplified)
function toGujaratiScript(text) {
  let out = "";
  let t = text.toLowerCase();

  for (let i = 0; i < t.length; i++) {
    let two = t[i] + (t[i+1] || "");
    if (map[two]) {
      out += map[two];
      i++;
    } else if (map[t[i]]) {
      out += map[t[i]];
    } else {
      out += t[i];
    }
  }
  return out;
}

function translate() {
  const input = document.getElementById("input").value;

  document.getElementById("roman").innerText = toRomanGujarati(input);
  document.getElementById("gujarati").innerText = toGujaratiScript(input);
}
